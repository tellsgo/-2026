         def _draw(self):
             """绘制游戏画面"""
             self.stdscr.clear()

             # 绘制边界
             for x in range(self.width):
                 self.stdscr.addch(0, x, curses.ACS_HLINE)
                 self.stdscr.addch(self.playable_height - 1, x, curses.ACS_HLINE)

             for y in range(1, self.playable_height - 1):
                 self.stdscr.addch(y, 0, curses.ACS_VLINE)
                 self.stdscr.addch(y, self.width - 1, curses.ACS_VLINE)

             # 绘制角落
             self.stdscr.addch(0, 0, curses.ACS_ULCORNER)
             self.stdscr.addch(0, self.width - 1, curses.ACS_URCORNER)
             self.stdscr.addch(self.playable_height - 1, 0, curses.ACS_LLCORNER)
             self.stdscr.addch(self.playable_height - 1, self.width - 1, curses.ACS_LRCORNER)

             # 绘制蛇
             for i, (y, x) in enumerate(self.snake):
                 if i == 0:
                     # 蛇头
                     self.stdscr.addch(y, x, 'O', curses.color_pair(1) | curses.A_BOLD)
                 else:
                     # 蛇身
                     self.stdscr.addch(y, x, 'o', curses.color_pair(2))

             # 绘制食物
             food_y, food_x = self.food
             self.stdscr.addch(food_y, food_x, '*', curses.color_pair(3) | curses.A_BOLD)

             # 绘制分数

- 问题与解决：1. 安装失败：优先检查系统版本与虚拟化设置。
             2. 下载慢：切换网络或使用国内镜像。
             3. 图形界面不可用：确认 WSLg 支持与 Windows 更新状态。
             4. 无法将磁盘“C:\Program Files\WSL\system.vhd”附加到 WSL2,系统找不到指定的文件: 因vhdx文件系统重启后被自动删除，可先检查并清理残留，再重新运行安装程
                 或：
                 wsl --export Ubuntu-22.04 D:\backup\ubuntu.tar
                 wsl --unregister Ubuntu-22.04
                 wsl --import Ubuntu-22.04 D:\wsl\ D:\backup\ubuntu.tar --version 2
             5. 若游戏运行时界面闪烁频繁：可能是刷新率问题
             6. vscode中多条代码报错，但游戏仍可以在终端正常运行：可能是因为VSCode中使用的Python解释器与你用于运行程序的终端中的Python解释器不一致，检查是否正确激活了虚拟环境。可以在VSCode底部的状态栏                  查看当前的Python解释器，并确保它指向正确的虚拟环境。或是依赖库未安装，在VSCode中打开项目的命令面板（可以通过按Ctrl+Shift+P或Cmd+Shift+P调出），安装项目所需的依赖库：
                pip install -r requirements.tx
